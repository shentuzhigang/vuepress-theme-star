<CodeGroup>
  <CodeGroupItem title="YARN" active>

```bash
# 在你的项目中安装
yarn add - D @starzkg/vuepress-theme-star

# 新建一个 markdown 文件
echo '# Hello VuePress Theme Star' > README.md

# 开始写作
yarn vuepress dev

# 构建静态文件
yarn vuepress build
```

  </CodeGroupItem>

  <CodeGroupItem title="NPM">

```bash
# 在你的项目中安装
npm install -D @starzkg/vuepress-theme-star

# 新建一个 markdown 文件
echo '# Hello VuePress Theme Star' > README.md

# 开始写作
npx vuepress dev

# 构建静态文件
npx vuepress build
```
